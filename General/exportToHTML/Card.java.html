1    /**

5    public class Card

15       }

18        @param name1 the given name
19        */
20       public Card(String name1)

23       }
24   
25       public void changeName (String n){name = n;}
26       /**
27        Accessor method for the name
28        @return the name
29        */
30       public String getName()

33       }

35        Tests whether the card is expired.

38       public boolean isExpired()

40           return false;
41       }

46       public String format()

49       }
50       /**

55        @return string denoting the object
56        */
57       public String toString()
58       {
59           return getClass().getName() + "[name = "
60                   + name + "]" ;
61       }
62       /**
63        Returns true if two cards have the same name and class.
64        Returns false if one card is null or not the same class.
65        Note the fancy way of checking the name of the class.  Here
66        we know the class name is Card, but when subclasses run this
67        method we want the subclass name.
68        @return true if the two objects are same type and have same
69        instance variable values
70        */
71       public boolean equals(Object other)
72       {
73           if (other == null) return false ;
74           if (! getClass().equals(other.getClass())) return false ;
75           Card card = (Card) other ;
76           return name.equals(card.name) ;
